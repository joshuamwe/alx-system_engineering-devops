#!/usr/bin/env bash
# displays bad luck for 4th iteration
# displays good luck for 8th iteration
# displays Best School for other iterations

i=1
while ((i <= 10))
do
    if ((i == 4))
    then
	echo "bad luck"
    elif ((i == 8))
    then
	echo "good luck"
    else
	echo "Best School"
    fi
    ((i++))
done
